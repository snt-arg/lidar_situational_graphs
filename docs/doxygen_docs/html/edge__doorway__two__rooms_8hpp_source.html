51  void computeError() override {
52  const VertexDoorWay* v1 = static_cast<const VertexDoorWay*>(_vertices[0]);
53  const VertexDoorWay* v2 = static_cast<const VertexDoorWay*>(_vertices[1]);
54  const VertexRoom* v3 = static_cast<const VertexRoom*>(_vertices[2]);
55  const VertexRoom* v4 = static_cast<const VertexRoom*>(_vertices[3]);
56 
57  Eigen::Isometry3d doorway1_coord_r1 = v1->estimate();
58  Eigen::Isometry3d doorway1_coord_r2 = v2->estimate();
59  Eigen::Isometry3d room1_coord_w = v3->estimate();
60  Eigen::Isometry3d room2_coord_w = v4->estimate();
61 
62  // Convert room 1 center to room pose (Transformation matrix)
63  Eigen::Matrix4d room1_pose_w;
64  Eigen::Vector3d room1_trans;
65  room1_pose_w.setIdentity();
66  room1_trans.x() = room1_coord_w.translation().x();
67  room1_trans.y() = room1_coord_w.translation().y();
68  room1_trans.z() = room1_coord_w.translation().z();
69  room1_pose_w.topRightCorner<3, 1>() = room1_trans;
70 
71  // Convert door1 position vector to 4x1
72  Eigen::Vector4d doorway1_pose_r1(doorway1_coord_r1.translation().x(),
73  doorway1_coord_r1.translation().y(),
74  doorway1_coord_r1.translation().z(),
75  1);
76 
77  // Convert room 2 center to room pose (Transformation matrix)
78  Eigen::Matrix4d room2_pose_w;
79  Eigen::Vector3d room2_trans;
80  room2_pose_w.setIdentity();
81  room2_trans.x() = room2_coord_w.translation().x();
82  room2_trans.y() = room2_coord_w.translation().y();
83  room2_trans.z() = room2_coord_w.translation().z();
84  room2_pose_w.topRightCorner<3, 1>() = room2_trans;
85 
86  // Convert door1 position vector to 4x1
87  Eigen::Vector4d doorway1_pose_r2(doorway1_coord_r2.translation().x(),
88  doorway1_coord_r2.translation().y(),
89  doorway1_coord_r2.translation().z(),
90  1);
91 
92  // Calculate error
93  Eigen::Vector4d diff =
94  (room1_pose_w * doorway1_pose_r1) - (room2_pose_w * doorway1_pose_r2);
95  Eigen::Vector3d error(diff.x(), diff.y(), diff.z());
96  _error = error;
97  }
