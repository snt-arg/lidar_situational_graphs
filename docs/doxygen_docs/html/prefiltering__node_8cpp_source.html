181  void cloud_callback(const sensor_msgs::msg::PointCloud2::SharedPtr src_cloud_msg) {
182  pcl::PointCloud<PointT>::Ptr src_cloud(new pcl::PointCloud<PointT>());
183  pcl::fromROSMsg(*src_cloud_msg, *src_cloud);
184 
185  if (src_cloud->empty()) {
186  return;
187  }
188 
189  src_cloud = deskewing(src_cloud);
190 
191  // if base_link_frame is defined, transform the input cloud to the frame
192  geometry_msgs::msg::TransformStamped transform_msg;
193  if (!base_link_frame.empty()) {
194  try {
195  transform_msg = tf_buffer->lookupTransform(
196  base_link_frame, src_cloud->header.frame_id, tf2::TimePointZero);
197  } catch (const tf2::TransformException& ex) {
198  RCLCPP_INFO(this->get_logger(),
199  "Could not transform %s to %s: %s",
200  base_link_frame.c_str(),
201  src_cloud->header.frame_id.c_str(),
202  ex.what());
203  return;
204  }
205 
206  pcl::PointCloud<PointT>::Ptr transformed(new pcl::PointCloud<PointT>());
207  pcl_ros::transformPointCloud(*src_cloud, *transformed, transform_msg);
208  transformed->header.frame_id = base_link_frame;
209  transformed->header.stamp = src_cloud->header.stamp;
210  src_cloud = transformed;
211  }
212 
213  pcl::PointCloud<PointT>::ConstPtr filtered = distance_filter(src_cloud);
214  filtered = downsample(filtered);
215  filtered = outlier_removal(filtered);
216 
217  sensor_msgs::msg::PointCloud2 filtered_msg;
218  pcl::toROSMsg(*filtered, filtered_msg);
219  points_pub->publish(filtered_msg);
220  }

250  pcl::PointCloud<PointT>::ConstPtr distance_filter(
251  const pcl::PointCloud<PointT>::ConstPtr& cloud) const {
252  pcl::PointCloud<PointT>::Ptr filtered(new pcl::PointCloud<PointT>());
253  filtered->reserve(cloud->size());
254 
255  std::copy_if(cloud->begin(),
256  cloud->end(),
257  std::back_inserter(filtered->points),
258  [&](const PointT& p) {
259  double d = p.getVector3fMap().norm();
260  return d > distance_near_thresh && d < distance_far_thresh;
261  });
262 
263  filtered->width = filtered->size();
264  filtered->height = 1;
265  filtered->is_dense = false;
266 
267  filtered->header = cloud->header;
268 
269  return filtered;
270  }

272  pcl::PointCloud<PointT>::Ptr deskewing(const pcl::PointCloud<PointT>::Ptr& cloud) {
273  rclcpp::Time stamp = pcl_conversions::fromPCL(cloud->header.stamp);
274  if (imu_queue.empty()) {
275  return cloud;
276  }
277 
278  // the color encodes the point number in the point sequence
279  if (colored_pub->get_subscription_count() > 0) {
280  pcl::PointCloud<pcl::PointXYZRGB>::Ptr colored(
281  new pcl::PointCloud<pcl::PointXYZRGB>());
282  colored->header = cloud->header;
283  colored->is_dense = cloud->is_dense;
284  colored->width = cloud->width;
285  colored->height = cloud->height;
286  colored->resize(cloud->size());
287 
288  for (size_t i = 0; i < cloud->size(); i++) {
289  double t = static_cast<double>(i) / cloud->size();
290  colored->at(i).getVector4fMap() = cloud->at(i).getVector4fMap();
291  colored->at(i).r = 255 * t;
292  colored->at(i).g = 128;
293  colored->at(i).b = 255 * (1 - t);
294  }
295 
296  sensor_msgs::msg::PointCloud2 colored_msg;
297  pcl::toROSMsg(*cloud, colored_msg);
298  colored_pub->publish(colored_msg);
299  }
300 
301  sensor_msgs::msg::Imu::SharedPtr imu_msg = imu_queue.front();
302 
303  auto loc = imu_queue.begin();
304  for (; loc != imu_queue.end(); loc++) {
305  imu_msg = (*loc);
306 
307  if (rclcpp::Time((*loc)->header.stamp) > stamp) {
308  break;
309  }
310  }
311 
312  imu_queue.erase(imu_queue.begin(), loc);
313 
314  Eigen::Vector3f ang_v(imu_msg->angular_velocity.x,
315  imu_msg->angular_velocity.y,
316  imu_msg->angular_velocity.z);
317  ang_v *= -1;
318 
319  pcl::PointCloud<PointT>::Ptr deskewed(new pcl::PointCloud<PointT>());
320  deskewed->header = cloud->header;
321  deskewed->is_dense = cloud->is_dense;
322  deskewed->width = cloud->width;
323  deskewed->height = cloud->height;
324  deskewed->resize(cloud->size());
325 
326  double scan_period =
327  this->get_parameter("scan_period").get_parameter_value().get<double>();
328  for (size_t i = 0; i < cloud->size(); i++) {
329  const auto& pt = cloud->at(i);
330 
331  // TODO: transform IMU data into the LIDAR frame
332  double delta_t = scan_period * static_cast<double>(i) / cloud->size();
333  Eigen::Quaternionf delta_q(1,
334  delta_t / 2.0 * ang_v[0],
335  delta_t / 2.0 * ang_v[1],
336  delta_t / 2.0 * ang_v[2]);
337  Eigen::Vector3f pt_ = delta_q.inverse() * pt.getVector3fMap();
338 
339  deskewed->at(i) = cloud->at(i);
340  deskewed->at(i).getVector3fMap() = pt_;
341  }
342 
343  return deskewed;
344  }
