286  Eigen::Matrix4f matching(const rclcpp::Time& stamp,
287  const pcl::PointCloud<PointT>::ConstPtr& cloud) {
288  if (!keyframe) {
289  prev_time = rclcpp::Time();
290  prev_trans.setIdentity();
291  keyframe_pose.setIdentity();
292  keyframe_stamp = stamp;
293  keyframe = downsample(cloud);
294  registration->setInputTarget(keyframe);
295  return Eigen::Matrix4f::Identity();
296  }
297 
298  auto filtered = downsample(cloud);
299  registration->setInputSource(filtered);
300 
301  std::string msf_source;
302  Eigen::Isometry3f msf_delta = Eigen::Isometry3f::Identity();
303 
304  if (this->get_parameter("enable_imu_frontend").get_parameter_value().get<bool>()) {
305  if (msf_pose && rclcpp::Time(msf_pose->header.stamp) > keyframe_stamp &&
306  msf_pose_after_update &&
307  rclcpp::Time(msf_pose_after_update->header.stamp) > keyframe_stamp) {
308  Eigen::Isometry3d pose0 = pose2isometry(msf_pose_after_update->pose.pose);
309  Eigen::Isometry3d pose1 = pose2isometry(msf_pose->pose.pose);
310  Eigen::Isometry3d delta = pose0.inverse() * pose1;
311 
312  msf_source = "imu";
313  msf_delta = delta.cast<float>();
314  } else {
315  std::cerr << "msf data is too old" << std::endl;
316  }
317  } else if (this->get_parameter("enable_robot_odometry_init_guess")
318  .get_parameter_value()
319  .get<bool>()) {
320  geometry_msgs::msg::TransformStamped transform_msg;
321  if (tf_buffer->canTransform(cloud->header.frame_id,
322  stamp,
323  cloud->header.frame_id,
324  prev_time,
325  robot_odom_frame_id)) {
326  try {
327  transform_msg = tf_buffer->lookupTransform(cloud->header.frame_id,
328  stamp,
329  cloud->header.frame_id,
330  prev_time,
331  robot_odom_frame_id);
332  msf_source = "odometry";
333  msf_delta = tf2isometry(transform_msg).cast<float>();
334  } catch (const tf2::TransformException& ex) {
335  RCLCPP_INFO(this->get_logger(),
336  "Could not transform %s to %s: %s",
337  cloud->header.frame_id.c_str(),
338  robot_odom_frame_id.c_str(),
339  ex.what());
340  }
341  } else if (tf_buffer->canTransform(
342  cloud->header.frame_id,
343  rclcpp::Time(0, 0, this->get_clock()->get_clock_type()),
344  cloud->header.frame_id,
345  prev_time,
346  robot_odom_frame_id)) {
347  transform_msg = tf_buffer->lookupTransform(
348  cloud->header.frame_id,
349  rclcpp::Time(0, 0, this->get_clock()->get_clock_type()),
350  cloud->header.frame_id,
351  prev_time,
352  robot_odom_frame_id);
353  msf_source = "odometry";
354  msf_delta = tf2isometry(transform_msg).cast<float>();
355  }
356  }
357 
358  pcl::PointCloud<PointT>::Ptr aligned(new pcl::PointCloud<PointT>());
359  registration->align(*aligned, prev_trans * msf_delta.matrix());
360 
361  publish_scan_matching_status(
362  stamp, cloud->header.frame_id, aligned, msf_source, msf_delta);
363 
364  if (!registration->hasConverged()) {
365  RCLCPP_INFO(this->get_logger(), "scan matching has not converged!!");
366  RCLCPP_INFO(this->get_logger(),
367  "ignore this frame: %s ",
368  std::to_string(stamp.seconds()).c_str());
369  return keyframe_pose * prev_trans;
370  }
371 
372  Eigen::Matrix4f trans = registration->getFinalTransformation();
373  Eigen::Matrix4f odom = keyframe_pose * trans;
374 
375  if (transform_thresholding) {
376  Eigen::Matrix4f delta = prev_trans.inverse() * trans;
377  double dx = delta.block<3, 1>(0, 3).norm();
378  double da = std::acos(Eigen::Quaternionf(delta.block<3, 3>(0, 0)).w());
379 
380  if (dx > max_acceptable_trans || da > max_acceptable_angle) {
381  RCLCPP_INFO(
382  this->get_logger(), "too large transform!! %f [m], %f [rad]", dx, da);
383  RCLCPP_INFO(this->get_logger(),
384  "ignore this frame %s ",
385  std::to_string(stamp.seconds()).c_str());
386  return keyframe_pose * prev_trans;
387  }
388  }
389 
390  prev_time = stamp;
391  prev_trans = trans;
392 
393  auto keyframe_trans =
394  matrix2transform(stamp, keyframe_pose, odom_frame_id, "keyframe");
395  keyframe_broadcaster->sendTransform(keyframe_trans);
396 
397  double delta_trans = trans.block<3, 1>(0, 3).norm();
398  double delta_angle = std::acos(Eigen::Quaternionf(trans.block<3, 3>(0, 0)).w());
399  double delta_time = (stamp - keyframe_stamp).seconds();
400  if (delta_trans > keyframe_delta_trans || delta_angle > keyframe_delta_angle ||
401  delta_time > keyframe_delta_time) {
402  keyframe = filtered;
403  registration->setInputTarget(keyframe);
404 
405  keyframe_pose = odom;
406  keyframe_stamp = stamp;
407  prev_time = stamp;
408  prev_trans.setIdentity();
409  }
410 
411  if (aligned_points_pub->get_subscription_count() > 0) {
412  pcl::transformPointCloud(*cloud, *aligned, odom);
413  sensor_msgs::msg::PointCloud2 aligned_msg;
414  pcl::toROSMsg(*aligned, aligned_msg);
415  aligned_msg.header.frame_id = odom_frame_id;
416  aligned_points_pub->publish(aligned_msg);
417  }
418 
419  return odom;
420  }

427  void publish_odometry(const rclcpp::Time& stamp,
428  const std::string& base_frame_id,
429  const Eigen::Matrix4f& pose) {
430  // publish transform stamped for IMU integration
431  geometry_msgs::msg::TransformStamped odom_trans =
432  matrix2transform(stamp, pose, odom_frame_id, base_frame_id);
433  trans_pub->publish(odom_trans);
434 
435  // broadcast the transform over tf
436  if (publish_tf) odom_broadcaster->sendTransform(odom_trans);
437 
438  // publish the transform
439  nav_msgs::msg::Odometry odom;
440  odom.header.stamp = stamp;
441  odom.header.frame_id = odom_frame_id;
442 
443  odom.pose.pose.position.x = pose(0, 3);
444  odom.pose.pose.position.y = pose(1, 3);
445  odom.pose.pose.position.z = pose(2, 3);
446  odom.pose.pose.orientation = odom_trans.transform.rotation;
447 
448  odom.child_frame_id = base_frame_id;
449  odom.twist.twist.linear.x = 0.0;
450  odom.twist.twist.linear.y = 0.0;
451  odom.twist.twist.angular.z = 0.0;
452 
453  odom_pub->publish(odom);
454  }

459  void publish_scan_matching_status(const rclcpp::Time& stamp,
460  const std::string& frame_id,
461  pcl::PointCloud<pcl::PointXYZI>::ConstPtr aligned,
462  const std::string& msf_source,
463  const Eigen::Isometry3f& msf_delta) {
464  if (status_pub->get_subscription_count() == 0) {
465  return;
466  }
467 
468  s_graphs::msg::ScanMatchingStatus status;
469  status.header.frame_id = frame_id;
470  status.header.stamp = stamp;
471  status.has_converged = registration->hasConverged();
472  status.matching_error = registration->getFitnessScore();
473 
474  const double max_correspondence_dist = 0.5;
475 
476  int num_inliers = 0;
477  std::vector<int> k_indices;
478  std::vector<float> k_sq_dists;
479  for (size_t i = 0; i < aligned->size(); i++) {
480  const auto& pt = aligned->at(i);
481  registration->getSearchMethodTarget()->nearestKSearch(
482  pt, 1, k_indices, k_sq_dists);
483  if (k_sq_dists[0] < max_correspondence_dist * max_correspondence_dist) {
484  num_inliers++;
485  }
486  }
487  status.inlier_fraction = static_cast<float>(num_inliers) / aligned->size();
488 
489  status.relative_pose = isometry2pose(
490  Eigen::Isometry3f(registration->getFinalTransformation()).cast<double>());
491 
492  if (!msf_source.empty()) {
493  status.prediction_labels.resize(1);
494  status.prediction_labels[0].data = msf_source;
495 
496  status.prediction_errors.resize(1);
497  Eigen::Isometry3f error =
498  Eigen::Isometry3f(registration->getFinalTransformation()).inverse() *
499  msf_delta;
500  status.prediction_errors[0] = isometry2pose(error.cast<double>());
501  }
502 
503  status_pub->publish(status);
504  }
