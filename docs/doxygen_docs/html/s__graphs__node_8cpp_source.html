484  void raw_odom_callback(const nav_msgs::msg::Odometry::SharedPtr odom_msg) {
485  Eigen::Isometry3d odom = odom2isometry(odom_msg);
486  geometry_msgs::msg::TransformStamped odom2map_transform;
487  Eigen::Isometry3d map2map_trans(trans_map2map.cast<double>());
488  Eigen::Matrix4f odom_corrected;
489  if (use_map2map_transform) {
490  Eigen::Isometry3d odom_trans = map2map_trans * odom;
491  odom_corrected = trans_odom2map * odom_trans.matrix().cast<float>();
492  } else {
493  odom_corrected = trans_odom2map * odom.matrix().cast<float>();
494  }
495  geometry_msgs::msg::PoseStamped pose_stamped_corrected =
496  matrix2PoseStamped(odom_msg->header.stamp, odom_corrected, map_frame_id);
