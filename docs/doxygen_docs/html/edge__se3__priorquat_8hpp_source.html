39 void computeError() override {
40 const g2o::VertexSE3* v1 = static_cast<const g2o::VertexSE3*>(_vertices[0]);
41
42 Eigen::Quaterniond estimate = Eigen::Quaterniond(v1->estimate().linear());
43
44 if(_measurement.coeffs().dot(estimate.coeffs()) < 0.0) {
45 estimate.coeffs() = -estimate.coeffs();
46 }
47 _error = estimate.vec() - _measurement.vec();
48 }
49
50 void setMeasurement(const Eigen::Quaterniond& m) override {
51 _measurement = m;
52 if(m.w() < 0.0) {
53 _measurement.coeffs() = -m.coeffs();
54 }
55 }

57 virtual bool read(std::istream& is) override {

60 setMeasurement(q);
61 for(int i = 0; i < information().rows(); ++i)
62 for(int j = i; j < information().cols(); ++j) {
63 is >> information()(i, j);
64 if(i != j) information()(j, i) = information()(i, j);
