101  bool contains(std::vector<T> vec, const T& elem) {
102  bool result = false;

124  void map_planes_callback(const s_graphs::msg::PlanesData::SharedPtr map_planes_msg) {

130  void flush_map_planes(std::vector<s_graphs::msg::PlaneData>& current_x_vert_planes,
131  std::vector<s_graphs::msg::PlaneData>& current_y_vert_planes) {
132  std::lock_guard<std::mutex> lock(map_plane_mutex);
133  for (const auto& x_map_planes_msg : x_vert_plane_queue) {
134  for (const auto& x_map_plane : x_map_planes_msg) {
135  if (!contains(current_x_vert_planes, x_map_plane) ||
136  current_x_vert_planes.empty()) {
137  current_x_vert_planes.push_back(x_map_plane);
138  } else {
139  continue;
140  }
141  }
142  x_vert_plane_queue.pop_front();
143  }
144 
145  for (const auto& y_map_planes_msg : y_vert_plane_queue) {
146  for (const auto& y_map_plane : y_map_planes_msg) {
147  if (!contains(current_y_vert_planes, y_map_plane) ||
148  current_y_vert_planes.empty()) {
149  current_y_vert_planes.push_back(y_map_plane);
150  } else {
151  continue;
152  }
153  }
154  y_vert_plane_queue.pop_front();
155  }
156  }

158  void floor_plan_callback() {
159  std::vector<s_graphs::msg::PlaneData> current_x_vert_planes, current_y_vert_planes;
160  flush_map_planes(current_x_vert_planes, current_y_vert_planes);
161 
162  if (current_x_vert_planes.empty() && current_y_vert_planes.empty()) {
163  // RCLCPP_INFO(this->get_logger(), "Did not receive any mapped planes");
164  return;
165  }
166 
167  auto t1 = this->now();
168  // extract_rooms(current_x_vert_planes, current_y_vert_planes);
169  extract_floors(current_x_vert_planes, current_y_vert_planes);
170  auto t2 = this->now();
171  // std::cout << "duration to extract clusters: " << boost::format("%.3f") % (t2 -
172  // t1).toSec() << std::endl;
173  if (save_timings) {
174  time_recorder.open("/tmp/floor_seg_computation_time.txt",
175  std::ofstream::out | std::ofstream::app);
176  time_recorder << std::to_string((t2 - t1).seconds()) + " \n";
177  time_recorder.close();
178  }

181  void extract_rooms(
182  const std::vector<s_graphs::msg::PlaneData>& current_x_vert_planes,
183  const std::vector<s_graphs::msg::PlaneData>& current_y_vert_planes) {
184  std::vector<s_graphs::msg::RoomData> room_candidates_vec;
185  std::vector<pcl::PointCloud<pcl::PointXYZRGB>::Ptr> curr_cloud_clusters =
186  room_analyzer->extract_cloud_clusters();
187 
188  for (const auto& cloud_cluster : curr_cloud_clusters) {
189  pcl::PointCloud<pcl::PointXYZRGB>::Ptr cloud_hull(
190  new pcl::PointCloud<pcl::PointXYZRGB>);
191 
192  if (cloud_cluster->points.size() < 10) continue;
193 
194  float hull_area;
195  room_analyzer->extract_convex_hull(cloud_cluster, cloud_hull, hull_area);
196  if (hull_area < 1.5) {
197  std::cout << "subgraph area too small to be a room " << std::endl;
198  continue;
199  }
200 
201  visualization_msgs::msg::MarkerArray current_cloud_marker;
202  RoomInfo room_info = {
203  current_x_vert_planes, current_y_vert_planes, cloud_cluster};
204  room_analyzer->perform_room_segmentation(
205  room_info, cloud_cluster, room_candidates_vec, current_cloud_marker);
206 
207  s_graphs::msg::RoomsData room_candidates_msg;
208  room_candidates_msg.header.stamp = this->now();
209  room_candidates_msg.rooms = room_candidates_vec;
210  all_rooms_data_pub->publish(room_candidates_msg);
211  }
212  }

214  void extract_floors(
215  const std::vector<s_graphs::msg::PlaneData>& current_x_vert_planes,
216  const std::vector<s_graphs::msg::PlaneData>& current_y_vert_planes) {
217  std::vector<s_graphs::msg::PlaneData> floor_plane_candidates_vec;
218  floor_analyzer->perform_floor_segmentation(
219  current_x_vert_planes, current_y_vert_planes, floor_plane_candidates_vec);
220 
221  geometry_msgs::msg::Pose floor_center;
222  if (floor_plane_candidates_vec.size() == 4) {
223  floor_center = PlaneUtils::room_center(floor_plane_candidates_vec[0],
224  floor_plane_candidates_vec[1],
225  floor_plane_candidates_vec[2],
226  floor_plane_candidates_vec[3]);
227 
228  s_graphs::msg::RoomData floor_data_msg;
229  floor_data_msg.header.stamp = this->now();
230  floor_data_msg.id = 0;
231  floor_data_msg.x_planes.push_back(floor_plane_candidates_vec[0]);
232  floor_data_msg.x_planes.push_back(floor_plane_candidates_vec[1]);
233  floor_data_msg.y_planes.push_back(floor_plane_candidates_vec[2]);
234  floor_data_msg.y_planes.push_back(floor_plane_candidates_vec[3]);
235  floor_data_msg.room_center = floor_center;
236  floor_data_pub->publish(floor_data_msg);
237  }
238  }
